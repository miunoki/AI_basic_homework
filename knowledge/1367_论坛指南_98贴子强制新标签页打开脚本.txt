【标题】98贴子强制新标签页打开脚本

【帖子正文】
lz自己喜欢在新标签页打开贴子，98在某个版面下打开的贴子是默认当前标签页打开的，所以让AI搓了一个浏览器脚本。
不知道发哪个版面就发这里了。
```javascript
// ==UserScript==
// @name CC98 帖子标题新标签页打开
// @namespace http://tampermonkey.net/
// @version 1.0
// @description 强制 CC98 版面帖子列表中的标题链接在新标签页打开
// @author You
// @match https://www.cc98.org/board/*
// @grant none
// @run-at document-end
// ==/UserScript==

(function() {
 'use strict';

 // 主处理函数
 function processLinks() {
 // 构造选择器：匹配奇数/偶数行下的特定标题 div 中的所有链接
 const selector = 
 '.board-postItem-body-odd .board-postItem-title a, ' +
 '.board-postItem-body-even .board-postItem-title a';

 const links = document.querySelectorAll(selector);

 links.forEach(link => {
 // 设置新标签页打开
 link.setAttribute('target', '_blank');
 // 添加安全属性（防止新页面通过 window.opener 控制原页面）
 link.setAttribute('rel', 'noopener noreferrer');
 });
 }

 // 1. 页面加载完成后先执行一次
 processLinks();

 // 2. 监听页面动态变化（应对 AJAX 无限加载或翻页）
 const observer = new MutationObserver((mutations) => {
 // 检查是否有新节点加入
 const hasNewNodes = mutations.some(mutation => mutation.addedNodes.length > 0);
 if (hasNewNodes) {
 processLinks();
 }
 });

 // 开始监听整个 body 区域的 DOM 变化
 observer.observe(document.body, {
 childList: true,
 subtree: true
 });
})();
```
